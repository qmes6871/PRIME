// 1. 웹 앱 접속 시 화면을 렌더링하는 함수
function doGet(e) {
  const template = HtmlService.createTemplateFromFile('index');
  
  // URL 파라미터로 넘어온 이름과 연락처를 템플릿에 전달 (없으면 빈 칸)
  template.customerName = e.parameter.name || '';
  template.customerPhone = e.parameter.phone || '';
  
  return template.evaluate()
    .setTitle('프리미엄 보험 상담 사전 설문') // 브라우저 탭 이름
    .addMetaTag('viewport', 'width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

// 2. 고객이 제출한 데이터를 시트에 저장하는 함수
function submitSurveyData(formData) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName("고객데이터");
    
    // 배열 형태로 넘어온 다중 선택 항목을 콤마(,)로 연결
    const assetsStr = formData.assets ? formData.assets.join(', ') : '';
    const interestsStr = formData.interests ? formData.interests.join(', ') : '';
    
    // 시트 헤더 순서에 맞게 데이터 배열 생성
    const rowData = [
      new Date(),              // A: 접수일시
      formData.name,           // B: 고객명
      formData.phone,          // C: 연락처
      "알림톡 발송됨",           // D: 알림톡발송 (링크를 타고 왔으므로)
      formData.dob,            // E: 생년월일
      formData.job,            // F: 직업
      formData.address,        // G: 주소
      formData.family,         // H: 가족관계
      assetsStr,               // I: 보유현황
      formData.consultType,    // J: 희망상담
      interestsStr,            // K: 보험관심사
      formData.privacy ? "동의함" : "미동의", // L: 개인정보동의
      "신규접수"                 // M: 진행상태
    ];
    
    // 시트 마지막 줄에 데이터 추가
    sheet.appendRow(rowData);
    
    return { success: true, message: "상담 신청이 성공적으로 접수되었습니다." };
  } catch (error) {
    return { success: false, message: "오류가 발생했습니다: " + error.message };
  }
}

  <script>
    function handleFormSubmit(event) {
      event.preventDefault(); // 기본 폼 제출 방지
      
      const form = event.target;
      const submitBtn = document.getElementById('submitBtn');
      const btnText = document.getElementById('btnText');
      const btnLoader = document.getElementById('btnLoader');
      
      // 폼 데이터 수집
      const formData = {
        name: form.name.value,
        phone: form.phone.value,
        dob: form.dob.value,
        job: form.job.value,
        address: form.address.value,
        family: form.family.value,
        consultType: form.consultType.value,
        privacy: form.privacy.checked,
        // 다중 선택(체크박스) 데이터 수집
        assets: Array.from(form.querySelectorAll('input[name="assets"]:checked')).map(cb => cb.value),
        interests: Array.from(form.querySelectorAll('input[name="interests"]:checked')).map(cb => cb.value)
      };

      // 버튼 상태 변경 (로딩 중)
      submitBtn.disabled = true;
      btnText.style.display = 'none';
      btnLoader.style.display = 'block';

      // 서버(Code.gs)로 데이터 전송
      google.script.run
        .withSuccessHandler(function(response) {
          if(response.success) {
            alert('감사합니다! 성공적으로 접수되었습니다. 곧 연락드리겠습니다.');
            // 제출 성공 시 화면 초기화 또는 감사 페이지 이동 가능
            form.reset();
            submitBtn.disabled = false;
            btnText.style.display = 'block';
            btnLoader.style.display = 'none';
            btnText.innerText = "제출 완료되었습니다";
            submitBtn.style.backgroundColor = "#10b981"; // 성공 색상으로 변경
          } else {
            alert('오류: ' + response.message);
            resetButton();
          }
        })
        .withFailureHandler(function(error) {
          alert('서버와 통신 중 문제가 발생했습니다.');
          resetButton();
        })
        .submitSurveyData(formData);
        
      function resetButton() {
        submitBtn.disabled = false;
        btnText.style.display = 'block';
        btnLoader.style.display = 'none';
      }
    }
  </script>
