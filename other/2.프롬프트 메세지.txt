  // 생명/손해 카테고리 선택 시 보험사 목록 업데이트

  function toggleSections() {
    const type = document.getElementById('noticeType').value;
    const stdInputs = document.getElementById('standardInputs');
    const claimDocs = document.getElementById('claimDocsSection');
    const genDocs = document.getElementById('generalDocsSection');

    if (type === '보험금 청구서류 안내') {
      stdInputs.style.display = 'none';
      genDocs.style.display = 'none';
      claimDocs.style.display = 'block';
    } else {
      stdInputs.style.display = 'block';
      genDocs.style.display = 'block';
      claimDocs.style.display = 'none';
    }
  }

  function toggleGeneralDocs() {

  function formatPremium(input) {
    let value = input.value.replace(/[^0-9]/g, '');
    input.value = value.replace(/\B(?=(\d{3})+(?!\d))/g, ",");
  }

  function updatePhoneAndPreview() {

    document.getElementById('callCenter').value = phone;
    generatePreview();
  }

  // 핵심 텍스트 자동 완성 로직
  function generatePreview() {
    const name = document.getElementById('customerName').value || 'OOO';
    const type = document.getElementById('noticeType').value;
    const company = document.getElementById('insuranceCompany').value || 'OO보험사';
    const insType = document.getElementById('insuranceType').value || 'OO보험';
    const premium = document.getElementById('premium').value || 'O,OOO';
    const phone = document.getElementById('callCenter').value || 'OOOO-OOOO';
    
    let message = `안녕하세요. 16년차 보험전문가 김예준 본부장입니다.\n\n`;

    // 1. 항목별 안내 멘트 및 "추가로 OO할 수 있는 방법" 동적 멘트 적용
    if (type === '보험담당자 변경안내') {
      message += `${name} 고객님, 새롭게 고객님의 담당자가 되었습니다. 앞으로 꼼꼼한 관리 약속드립니다.\n`;
      message += `추가로 안전하게 담당자 변경 확인 및 점검을 진행할 수 있는 방법을 안내드립니다.\n\n`;
      message += `[변경 대상 보험 정보]\n`;
    } else if (type === '해지안내') {
      message += `${name} 고객님의 요청에 따른 해지 안내입니다.\n`;
      message += `추가로 원활하게 해지 처리를 하고 환급금을 수령할 수 있는 방법을 안내드립니다. 아래 보험사에 연락하시어 진행해 주시기 바랍니다.\n\n`;
      message += `[해지해야 할 보험 정보]\n`;
    } else if (type === '실효된보험 해지안내') {
      message += `${name} 고객님, 현재 실효 상태인 보험의 최종 해지 및 해약환급금 수령을 위한 안내입니다.\n`;

      message += `[실효된 보험 정보]\n`;
    } else if (type === '자동이체 해지안내') {
      message += `${name} 고객님, 보험 자동이체 해지 관련 안내입니다.\n`;
      message += `추가로 간편하게 자동이체를 해지할 수 있는 방법을 안내드립니다.\n`;
      message += `💡 안내: 자동이체 해지는 두 달 동안 납입이 되지 않아도, 정상적으로 보험 혜택을 받을 수 있습니다.\n\n`;
    } else if (type === '보험금 청구서류 안내') {
      message += `${name} 고객님, 요청하신 보험금 청구를 위한 안내입니다.\n`;
      message += `추가로 빠르고 정확하게 보험금을 청구할 수 있는 방법을 안내드립니다.\n\n`;
    } else {
      message += `${name} 고객님을 위한 맞춤형 안내입니다.\n\n`;
    }

    // 2. 입력 박스 정보 (청구 안내가 아닐 때만 포함)
    if (type !== '보험금 청구서류 안내' && type !== '') {
      message += `▪ 보험사 : ${company}\n`;
      message += `▪ 상품명 : ${insType}\n`;
      message += `▪ 보험료 : ${premium}원\n`;
      message += `📞 콜센터 : ${phone}\n\n`;

    // 3. 서류 리스트 삽입
    if (type === '보험금 청구서류 안내') {
      // 청구 서류 체크박스 내역 가져오기
      const checkedClaims = Array.from(document.querySelectorAll('#claimDocsGrid input:checked')).map(cb => cb.value);

        message += `[📌 필수 준비 서류 안내 (보험금 청구용)]\n`;
        checkedClaims.forEach(doc => { message += `- ${doc}\n`; });
        message += `\n`;
      } else {
        message += `(안내하실 청구 서류를 좌측에서 체크해 주세요)\n\n`;
      }
    } else {
      // 일반 서류 체크박스 내역 가져오기
      if (document.getElementById('docReq').checked) {
        const checkedGenerals = Array.from(document.querySelectorAll('#generalDocsGrid input:checked')).map(cb => cb.value);
        if(checkedGenerals.length > 0) {
          message += `[📌 추가 준비 서류 안내]\n`;
          checkedGenerals.forEach(doc => { message += `- ${doc}\n`; });
          message += `- (콜센터 상담원 안내에 따라 팩스로 전송해 주시면 됩니다.)\n\n`;
        }
      }
    }

    message += `진행하시면서 궁금한 점이 있으시거나, 제가 도와드릴 부분이 있다면 언제든지 편하게 답장 남겨주세요. 감사합니다.`;

    document.getElementById('previewText').value = message;
  }

  // 복사 및 카카오 버튼 (저장 버튼은 삭제됨!)
  function copyText() {

    if(!text) { alert('복사할 내용이 없습니다.'); return; }
    navigator.clipboard.writeText(text).then(() => {
      alert('메시지가 성공적으로 복사되었습니다. 카카오톡이나 문자에 붙여넣기 하세요!');
    });
  }
