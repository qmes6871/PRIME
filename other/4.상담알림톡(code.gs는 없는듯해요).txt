    // UI 상태 업데이트 함수
    function updateUI() {
      const type = templateSelect.value;
      
      // 섹션 초기화
      scheduleSection.style.display = 'none';
      infoSection.style.display = 'none';
      
      if (type === 'schedule') {
        scheduleSection.style.display = 'block';
        // 대면상담일 때만 장소 입력 표시
        meetSection.style.display = (consultMethod.value === '대면상담') ? 'block' : 'none';
      } else if (type === 'info') {
        infoSection.style.display = 'block';
      }
      
      renderMessage();
    }

    // 날짜 포맷팅 함수
    function formatDate(dateStr) {
      if (!dateStr) return "[일시 미정]";
      const d = new Date(dateStr);
      const yy = d.getFullYear().toString().slice(-2);
      const mm = d.getMonth() + 1;
      const dd = d.getDate();
      const hh = d.getHours();
      const min = d.getMinutes();
      const ampm = hh >= 12 ? '오후' : '오전';
      const displayHour = hh % 12 || 12;
      return `${yy}년 ${mm}월 ${dd}일 ${ampm} ${displayHour}시 ${min === 0 ? '00' : min}분`;
    }

          break;

        case 'schedule':
          const method = consultMethod.value;
          const dateStr = formatDate(consultDate.value);
          let placeStr = "";
          
          if (method === '대면상담') {
            const region = meetRegion.value;
            const detail = meetDetail.value || "[세부장소 미정]";
            placeStr = `\n■ 대면 장소 : [${region}] ${detail}`;
          }

          rawText = "[상담 일정 안내]\n프라임에셋 김예준 본부장입니다.\n" + name + " 고객님의 맞춤형 보험 상담 일정이 확정되었습니다.\n\n■ 상담 방식 : " + method + "\n■ 예정 일시 : " + dateStr + placeStr + "\n\n편안하고 유익한 상담을 위해 꼼꼼히 분석하여 준비하고 있겠습니다. 일정 변경이 필요하시면 언제든 편하게 말씀해 주세요!";
          break;

        case 'info':
          const selectedInfo = infoType.value;

          rawText = "[맞춤형 보험 정보 안내]\n" + name + " 고객님, 안녕하세요! 프라임에셋 김예준 본부장입니다.\n\n상담에 앞서, 고객님께서 궁금해하실 [" + selectedInfo.replace('안내', '') + "]에 대한 핵심 정보를 미리 준비했습니다. 아래 내용을 가볍게 확인해 보시면 앞으로의 상담에 큰 도움이 됩니다!\n\n" + links + "\n\n꼼꼼히 살펴보시고 궁금한 점은 상담 때 모두 속 시원히 해결해 드리겠습니다.😊";
          break;

        case 'finish':

          rawText = "https://litt.ly/withprime100\n\n저는 16년차 보험설계사이자\n50명의 전문 설계사를 총괄하는\n프라임에셋 김예준 본부장입니다.\n\n매년 1분기에는 고객분들의\n고객정보(주소변경, 직업변경, 검진)를\n꼼꼼하게 체크하여 변경을 도와드리고\n중요한 보험 이슈를 안내해드리고 있습니다\n\n보장분석 프로그램과 노션을 활용한\n전문 온라인 상담에 만족하셨다면\n많은 분들이 손해보지 않으시도록\n소개도 부탁드려요 :)\n\n상담에 친절하게 응해주셔서 감사드리고\n[보험금청구] 상황 발생시\n언제든 편하게 전화나 톡으로\n문의주세요😊";
          break;
      }

      outputArea.value = rawText;
    }

    // 이벤트 리스너 등록
    nameInput.addEventListener('input', renderMessage);
    templateSelect.addEventListener('change', updateUI);
    consultMethod.addEventListener('change', updateUI);
    consultDate.addEventListener('input', renderMessage);
    meetRegion.addEventListener('change', renderMessage);
    meetDetail.addEventListener('input', renderMessage);
    infoType.addEventListener('change', renderMessage);

    // 복사 기능
    function copyMessage() {
      if(!outputArea.value) {
        alert('먼저 발송 상황을 선택해주세요.');
        return;
      }
      outputArea.select();
      document.execCommand('copy');

    }
